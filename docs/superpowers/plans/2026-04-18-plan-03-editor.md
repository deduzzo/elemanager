# Plan 03 — Editor Vote Entry

**Goal:** Flusso mobile-first per editor/admin che inseriscono i voti reali di una sezione: aperture per giornata `open`, lista sezioni, form con tab per elezione (totali scheda + voti per lista + preferenze per candidato), autosave locale, stato draft/submitted.

**Tech:** React + TanStack Query + Supabase + Zustand (per local autosave state). Realtime già attivo.

**Prerequisiti:** Plan 01 + 02 (tag `plan-02-admin-complete`).

## Assunzioni UX

- **Editor vede solo giornate `stato='open'`**. RLS lato DB già permette insert risultati solo se `giornata_is_open`.
- **Admin vede tutto** (incluso draft/closed).
- **Form input numerici**: `inputMode="numeric"` + `pattern="[0-9]*"` per tastiera numerica nativa su mobile.
- **Autosave**: ogni 5 secondi salva draft in localStorage con chiave `elemanager:draft:<risultato_sezione_id o composite>`. Al reopen del form, se esiste draft locale non ancora submitted, lo propone.
- **Validazione soft**: `somma(preferenze_lista_X) ≤ voti_lista_X` → mostra warning (non blocca salvataggio bozza, blocca submit).
- **Navigazione**: bottom nav "Editor" → `/editor`.

## Routing

```
/editor                                         → EditorHomePage (lista giornate open)
/editor/giornate/:giornataId                    → SezioniPickPage (search + lista sezioni)
/editor/giornate/:giornataId/sezioni/:sezioneId → VoteEntryPage (form)
```

`/editor/*` protetto da `ProtectedRoute allow={['admin','editor']}`.

### risultati.ts

```ts
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { db } from './_db';
import type { RisultatoSezioneRow, RisultatoSezioneInsert, RisultatoSezioneUpdate } from '@/lib/database.types';
import { useRealtimeTable } from './useRealtimeTable';

const KEY = 'risultati';

export function useRisultatoPerSezioneElezione(sezioneId: string | undefined, elezioneId: string | undefined) {
  useRealtimeTable({
    table: 'risultati_sezione',
    invalidate: [[KEY, sezioneId, elezioneId]],
    enabled: !!sezioneId && !!elezioneId,
  });
  return useQuery({
    queryKey: [KEY, sezioneId, elezioneId],
    enabled: !!sezioneId && !!elezioneId,
    queryFn: async (): Promise<RisultatoSezioneRow | null> => {
      const { data, error } = await db
        .from('risultati_sezione')
        .select('*')
        .eq('sezione_id', sezioneId as string)
        .eq('elezione_id', elezioneId as string)
        .maybeSingle();
      if (error) throw error;
      return (data as RisultatoSezioneRow | null) ?? null;
    },
  });
}

export function useUpsertRisultato() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async (input: RisultatoSezioneInsert | (RisultatoSezioneUpdate & { id?: string; sezione_id: string; elezione_id: string })) => {
      const { data, error } = await db
        .from('risultati_sezione')
        .upsert(input as RisultatoSezioneInsert, { onConflict: 'sezione_id,elezione_id' })
        .select()
        .single();
      if (error) throw error;
      return data as RisultatoSezioneRow;
    },
    onSuccess: (row) => {
      qc.invalidateQueries({ queryKey: [KEY] });
      qc.invalidateQueries({ queryKey: [KEY, row.sezione_id, row.elezione_id] });
    },
  });
}

export function useRisultatiByGiornata(giornataId: string | undefined) {
  // Usa join via query: SELECT * FROM risultati JOIN sezioni ON sezione_id = sezioni.id WHERE sezioni.giornata_id = giornataId
  useRealtimeTable({ table: 'risultati_sezione', invalidate: [[KEY, 'giornata', giornataId]], enabled: !!giornataId });
  return useQuery({
    queryKey: [KEY, 'giornata', giornataId],
    enabled: !!giornataId,
    queryFn: async (): Promise<RisultatoSezioneRow[]> => {
      const { data, error } = await db
        .from('risultati_sezione')
        .select('*, sezione:sezioni!inner(giornata_id)')
        .eq('sezione.giornata_id', giornataId as string);
      if (error) throw error;
      return ((data ?? []) as unknown) as RisultatoSezioneRow[];
    },
  });
}
```

### votiLista.ts

```ts
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
import { db } from './_db';
import type { VotoListaRow } from '@/lib/database.types';
import { useRealtimeTable } from './useRealtimeTable';

const KEY = 'voti-lista';

export function useVotiListaByRisultato(risultatoId: string | undefined) {
  useRealtimeTable({ table: 'voti_lista', invalidate: [[KEY, risultatoId]], enabled: !!risultatoId });
  return useQuery({
    queryKey: [KEY, risultatoId],
    enabled: !!risultatoId,
    queryFn: async (): Promise<VotoListaRow[]> => {
      const { data, error } = await db
        .from('voti_lista')
        .select('*')
        .eq('risultato_sezione_id', risultatoId as string);
      if (error) throw error;
      return (data ?? []) as VotoListaRow[];
    },
  });
}

export function useUpsertVotiLista() {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: async ({ risultatoId, rows }: { risultatoId: string; rows: Array<{ lista_id: string; voti: number }> }) => {
      const payload = rows.map((r) => ({ ...r, risultato_sezione_id: risultatoId }));
      const { data, error } = await db
        .from('voti_lista')
        .upsert(payload, { onConflict: 'risultato_sezione_id,lista_id' })
        .select();
      if (error) throw error;
      return (data ?? []) as VotoListaRow[];
    },
    onSuccess: (_d, v) => qc.invalidateQueries({ queryKey: [KEY, v.risultatoId] }),
  });
}
```

### useAutosave

```ts
import { useEffect } from 'react';

export function useAutosave<T>(key: string, value: T, delayMs = 5000) {
  useEffect(() => {
    const t = setTimeout(() => {
      try { localStorage.setItem(key, JSON.stringify({ value, ts: Date.now() })); } catch {}
    }, delayMs);
    return () => clearTimeout(t);
  }, [key, value, delayMs]);
}
export function readAutosave<T>(key: string): { value: T; ts: number } | null {
  try {
    const raw = localStorage.getItem(key);
    return raw ? JSON.parse(raw) : null;
  } catch { return null; }
}
export function clearAutosave(key: string) {
  try { localStorage.removeItem(key); } catch {}
}
```
